function updateTabCounts() { let qaOpen = 0, qaTotal = 0; let partnershipsOpen = 0, partnershipsTotal = 0; Object.values(appData).forEach(app => { const hasQaIssues = app.teams?.includes('QA') && app.qa_issues?.length > 0; const hasPartnershipsIssues = app.teams?.includes('Partnerships') && app.partnerships_issues?.length > 0; // Count QA apps - open vs total if (hasQaIssues) { qaTotal++; const hasUnsolvedQA = app.qa_issues.some(issue => !isIssueSolved(app.app_name, issue, 'QA')); if (hasUnsolvedQA) qaOpen++; } // Count Partnerships apps - open vs total if (hasPartnershipsIssues) { partnershipsTotal++; const hasUnsolvedPartnerships = app.partnerships_issues.some(issue => !isIssueSolved(app.app_name, issue, 'Partnerships')); if (hasUnsolvedPartnerships) partnershipsOpen++; } }); // All count = QA + Partnerships - overlap (apps with both) // Calculate unique apps with issues const allOpen = qaOpen + partnershipsOpen; const allTotal = qaTotal + partnershipsTotal; document.getElementById('tab-count-all').textContent = `(${allOpen}/${allTotal})`; document.getElementById('tab-count-qa').textContent = `(${qaOpen}/${qaTotal})`; document.getElementById('tab-count-partnerships').textContent = `(${partnershipsOpen}/${partnershipsTotal})`; }
